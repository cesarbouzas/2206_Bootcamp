-- public.candidate definition

-- Drop table

-- DROP TABLE public.candidate;

CREATE TABLE public.candidate (
	id serial4 NOT NULL,
	photo varchar(90) NULL,
	"name" varchar(255) NULL,
	surname varchar(255) NULL,
	birthday date NULL,
	dni varchar(255) NULL,
	phone varchar(255) NULL,
	email varchar(255) NULL,
	education int4 NULL,
	specialties varchar(255) NULL,
	status int4 NULL,
	origin int4 NULL,
	wage_level numeric NULL,
	experience_level int4 NULL,
	profile int4 NULL,
	"comment" varchar(255) NULL,
	linkedin varchar(255) NULL,
	CONSTRAINT candidate_fk_education FOREIGN KEY (education) REFERENCES public.education(educa_id),
	CONSTRAINT candidate_fk_experience_level FOREIGN KEY (experience_level) REFERENCES public.experience(exp_id),
	CONSTRAINT candidate_fk_origin FOREIGN KEY (origin) REFERENCES public.origin(ori_id),
	CONSTRAINT candidate_fk_profile FOREIGN KEY (profile) REFERENCES public.profile(prof_id),
	CONSTRAINT candidate_fk_status FOREIGN KEY (status) REFERENCES public.status(sta_id)
);
